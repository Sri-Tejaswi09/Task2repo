<script>
  function toggleTheme() {
    const html = document.documentElement;
    const btn = document.querySelector('.toggle-theme');
    const theme = html.getAttribute('data-theme');
    html.setAttribute('data-theme', theme === 'light' ? 'dark' : 'light');
    btn.textContent = theme === 'light' ? 'üåô' : '‚òÄ';
  }

  function toggleSidebar() {
    document.getElementById('sidebar').classList.toggle('hidden');
  }

  function addTask() {
    const input = document.getElementById('todoInput');
    const list = document.getElementById('todoList');
    const task = input.value.trim();
    if (task !== '') {
      const li = document.createElement('li');
      li.innerHTML = ${task} <button onclick="removeTask(this)">‚úñ</button>;
      list.appendChild(li);
      input.value = '';
    }
  }

  function removeTask(button) {
    button.parentElement.remove();
  }

  function addImage() {
    const url = document.getElementById('imageURL').value.trim();
    if (url) {
      const container = document.getElementById('galleryGrid');
      const item = document.createElement('div');
      item.className = 'gallery-item';
      item.innerHTML = `
        <img src="${url}" alt="Image" onerror="this.src='https://via.placeholder.com/150?text=Invalid+Image';" />
        <button onclick="removeImage(this)">‚úñ</button>
      `;
      container.appendChild(item);
      document.getElementById('imageURL').value = '';
    }
  }

  function removeImage(button) {
    button.parentElement.remove();
  }

  function handleSubmit(event, form) {
    event.preventDefault();
    form.reset();
    showSuccess();
  }

  function showSuccess() {
    const popup = document.getElementById('successPopup');
    popup.style.opacity = '1';

    for (let i = 0; i < 100; i++) {
      const confetti = document.createElement('div');
      confetti.classList.add('confetti');
      confetti.style.left = Math.random() * 100 + 'vw';
      confetti.style.backgroundColor = hsl(${Math.random() * 360}, 100%, 50%);
      confetti.style.animationDuration = (Math.random() * 2 + 2) + 's';
      document.body.appendChild(confetti);
      setTimeout(() => confetti.remove(), 3000);
    }

    setTimeout(() => {
      popup.style.opacity = '0';
    }, 3000);
  }
</script>
